subroutine paws()
character(len=1) :: letter
integer :: ios

   write(stdout,'(a)',advance='no',iostat=ios)'Enter [RETURN] to continue ...'
   read(*,'(a)',iostat=ios)letter
   write(stdout,'(a)')attr('<clear>')
   flush(unit=stdout,iostat=ios)

end subroutine paws
